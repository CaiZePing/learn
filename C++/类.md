		m_LogLevel = level;

# 构造函数

在创建一个实例的时候会调用构造函数，不管是在堆上，还是在栈上

## 基础的构造函数

```cpp
#include <iostream>

class Entity
{
public:
	float x,y;
	
	Entity()
	{
		this->x = 0.0f;
		this->y = 0.0f;
	}
	// 函数重载
	Entity(float x,float y)
	{
		this->x = x;
		this->y = y;
	}
	// 打印信息
	void Print()
	{
		std::cout << x << ", " << y << std::endl;
	}
};


int main()
{
	Entity e;
	e.Print(); // 输出 0, 0
	
	Entity e1(5.5f,6.6f);
	e1.Print(); // 输出 5.5, 6.6
	
	return 0;
}
```

## 拷贝构造函数

主要用于类成员变量中有指针的情况
防止浅拷贝所造成的多次释放同一块内存

```cpp
#include <iostream>

class String
{
public:
	friend std::ostream &operator<<(std::ostream &stream, const String &string);
private:
	char *m_Buffer;
	unsigned int m_Size;
public:
	String() = default;
	String(const char *string)
	{
		this->m_Size = strlen(string);
		this->m_Buffer = new char[this->m_Size + 1];
		memcpy(this->m_Buffer, string, this->m_Size);
		this->m_Buffer[this->m_Size] = 0;
	}
	// 构造拷贝函数
	String(const String &other)
		: m_Size(other.m_Size)
	{
		this->m_Buffer = new char[this->m_Size + 1];
		memcpy(this->m_Buffer, other.m_Buffer, this->m_Size + 1);
	}
	~String()
	{
		std::cout << "Destroyed String!" << std::endl;
		delete[] m_Buffer;
	}
	char &operator[](unsigned int index)
	{
		return this->m_Buffer[index];
	}
};

std::ostream &operator<<(std::ostream &stream, const String &string)
{
	stream << string.m_Buffer;
	return stream;
}

int main()
{
	String first = "CaiZeping";
	String second = first;
	
	second[3] = 'a';
	
	std::cout << first << std::endl;
	std::cout << second << std::endl;
	
	std::cin.get();
	return 0;
}
```

# 析构函数

会在实例生命周期结束的时候调用析构函数

## 基础的析构函数

```cpp
#include <iostream>

class Entity
{
public:
	float x,y;
	
	Entity()
	{
		this->x = 0.0f;
		this->y = 0.0f;
		std::cout << "Create Entity!" << std::endl;
	}
	
	~Entity()
	{
		std::cout << "Destroy Entity!" << std::endl;
	}
	
	// 打印信息
	void Print()
	{
		std::cout << x << ", " << y << std::endl;
	}
};

void Function()
{
	Entity e;
	e.Print(); // 输出 0, 0
}

int main()
{
	Function();
	/**
	 * 执行到这里的时候将会输出
	 * Create Entity!
	 * 0, 0
	 * Destroy Entity!
	 */
	return 0;
}
```

>[!Caution]
>⚠️析构函数只有一个，析构函数没有参数列表‼️

## 虚析构函数

```cpp
#include <iostream>

class Base
{
public:
	Base(){ std::cout << "Base Constructor\n"; }
	// 如果没有 virtual 在使用多态时，析构会出问题
	virtual ~Base(){ std::cout << "Base Destructor\n"; }
};

class Derives : public Base
{
public:
	Derives()
	{
		m_Array = new int[5];
		std::cout << "Derives Constructor\n"; 
	}
	~Derives()
	{
		delete[] m_Array;
		std::cout << "Derives Destructor\n";
	}
private:
	int* m_Array;
}

int main()
{
	// 没问题
	std::cout << "================" << std:endl;
	Base* base = new Base();
	delete base;
	std::cout << "================" << std:endl;
	Derives* derives = new Derives();
	delete derives;
	// 如果没有 virtual
	// der 将不会调用自己析构函数
	// 现在将会按照构造的相反顺序调用析构函数
	std::cout << "================" << std:endl;
	Base* der = new Derives();
	delete der;
	
	return 0;
}
```

# 虚函数

## 基础的虚函数

```cpp
#include <iostream>
#include <string>

class Entity
{
public:
	virtual std::string GetName(){return "Entity";}
};

class Player : public Entity
{
public:
	const std::string m_Name;
	Player(const std::string&name)
		: m_Name(name){}
	std::string GetName() override {return m_Name;}
};

void PrintName(Entity*entity)
{
	/**
	 * 这边将会调用不同的 GetName() 函数
	 * 如果基类中的 GetName() 函数之前不使用 virtual 关键字
	 * 修饰的话，就之后调用 Entity 的 GetName() 函数
	 * 最终的输出都是 Entity
	 */
	std::cout << entity->GetName() << std::endl;
}

int main()
{
	Entity e;
	Player p("Hello");
	
	PrintName(&e);
	PrintName(&p);
	
	return 0;
}
```

## 纯虚函数

类似接口，继承的子孙中，必须要将纯虚函数给实现，才可以实例化使用

```cpp
#include <iostream>
#include <string>

class Printable
{
public:
	// 最后必须得有 = 0，不然不是纯虚函数
	virtual std::string GetClassName() = 0;
};

class Entity : public Printable
{
public:
	virtual std::string GetName(){return "Entity";}
	std::string GetClassName() override {return "Entity";}
};

class Player : public Entity
{
public:
	const std::string m_Name;
	Player(const std::string&name)
		: m_Name(name){}
	std::string GetName() override {return m_Name;}
	std::string GetClassName() override {return "Player";}
};

class A : public Printable
{
public:
	std::string GetClassName() override {return "A";}
};

void Print(Printable*obj)
{
	std::cout << obj->GetClassName() << std::endl;
}

void PrintName(Entity*entity)
{
	std::cout << entity->GetName() << std::endl;
}

int main()
{
	Entity e;
	Player p("Hello");
	
	Print(&e);
	// 如果这个类中没有重写 GetClassName() 函数
	// 将会输出 Entity
	Print(&p);
	Print(new A());
	
	return 0;
}
```

# new 关键字

```cpp
#include <iostream>
#include <string>

class Entity
{
private:
	std::string m_Name;
public:
	Entity()
		: m_Name("Unknown")
	{
	}
	
	Entity(const std::string &name)
		: m_Name(name)
	{
	}
	
	const std::string GetName()const{return m_Name;}
};

int main()
{
	int a = 10;
	// 在堆上申请了一个连续的 50*sizeof(int) 字节大小的空间
	// 将内存的首地址返回
	int *b = new int[50];
	// 在堆上申请了一个连续的 1*sizeof(Entity) 字节大小的空间
	// 将内存的首地址返回
	Entity *e1 = new Entity;
	// placement new
	Entity *e2 = new(b) Entity;
	
	return 0;
	delete e1;
}
```

>[!Caution] 注意
>使用 `new` 关键字后，一定要记得使用 `delete` 释放这片空间
>如果使用 `new int[]` ，释放时需要使用 `delete[]`

# 运算符重载

```cpp
#include <iostream>

struct Vector2
{
	float x, y;
	Vector2(float x, float y)
		: x(x), y(y) {}
	// 加法
	Vector2 operator+(const Vector2 &other) const
	{
		return {this->x + other.x, this->y + other.y};
	}
	Vector2 Add(const Vector2 &other) const
	{
		return *this + other;
	}
	// 乘法
	Vector2 operator*(const Vector2 &other) const
	{
		return {this->x * other.x, this->y * other.y};
	}
	Vector2 Multiply(const Vector2 &other) const
	{
		return *this * other;
	}
	// 相等
	bool operator==(const Vector2 &other) const
	{
		return this->x == other.x && this->y == other.y;
	}
	bool Equal(const Vector2 &other) const
	{
		return *this == other;
	}
	// 不等
	bool operator!=(const Vector2 &other) const
	{
		return this->x != other.x || this->y != other.y;
	}
	bool NEqual(const Vector2 &other) const
	{
		return *this != other;
	}
};

std::ostream &operator<<(std::ostream &stream, const Vector2 &vector)
{
	stream << vector.x << ", " << vector.y;
	return stream;
}

int main()
{
	Vector2 position(2.3f, 34.34f);
	Vector2 speed(2.0f, 2.0f);
	Vector2 powerup(1.1f, 1.1f);
	
	Vector2 result = position + speed * powerup;
	std::cout << result << std::endl;
	
	return 0;
}
```

# 箭头操作符

## 用法1

```cpp
#include <iostream>

class Entity
{
private:
	int x;
public:
	Entity()
		: x(0){}
	Entity(int x)
		: x(x){}
	void Print()
	{
		std::cout << x << std::endl;
	}
	void Print() const
	{
		std::cout << x << std::endl;
	}
};

class ScopedPtr
{
private:
	Entity*m_Obj;
public:
	ScopedPtr(Entity*entity)
		: m_Obj(entity){}
	~ScopedPtr()
	{
		delete m_Obj;
	}
	Entity*operator->()
	{
		return m_Obj;
	}
	const Entity*operator->() const
	{
		return m_Obj;
	}
};

int main()
{
	ScopedPtr entity= new Entity();
	entity->Print();
	const ScopedPtr c_entity= new Entity(10);
	c_entity->Print();
	return 0;
}
```
