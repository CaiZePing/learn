# 隐式转换和 expece

```cpp
#include <iostream>
#include <string>

class Entity
{
private:
	std::string m_Name;
	int m_Age;
public:
	Entity()
		: m_Name("Unknown"),m_Age(-1)
	{
	}
	explicit Entity(const std::string &name)
		: m_Name(name),m_Age(-1)
	{
	}
	Entity(int age)
		: m_Name("unknown"),m_Age(age)
	{
	}
	const std::string GetName() const { return m_Name; }
};

void PrintEntity(const Entity& e)
{
	// print entity info
}

int main()
{
	// 这是不允许的
	// 因为这个构造使用了 explicit 关键字
	Entity e0 = "Cai";
	// 这是允许的
	Entity e1("Cai");
	// 这是可以的
	Entity e2 = 20;
	Entity e3(30);
	// 不可以
	PrintEntity("Cai");
	PrintEntity(std::string("Cai"));
	// 可以
	PrintEntity(20);
	PrintEntity(Entity(20));
	PrintEntity(Entity("Cai"));
	
	return 0;
}
```

# cast

|转换类型|适用场景|安全性|
|---|---|---|
|`static_cast`|基本类型转换、继承关系的向上转换|较高|
|`const_cast`|修改 const/volatile 属性|谨慎使用|
|`dynamic_cast`|多态类的向下转换（安全检查）|高（但有开销）|
|`reinterpret_cast`|底层二进制重解释（如指针与整数互转）|低（尽量避免）|

## static_cast

```cpp
int main()
{
	int a = 30;
	double d = static_cast<double>(a);
	return 0;
}
```

## dynamic_cast

```cpp
#include <iostream>

class Base
{
public:
	Base(){ std::cout << "Base Constructor\n"; }
	virtual ~Base(){ std::cout << "Base Destructor\n"; }
};

class BaseA : public Base
{
public:
	BaseA(){ std::cout << "BaseA Constructor\n"; }
	~BaseA(){ std::cout << "BaseA Destructor\n"; }
};

class BaseB : public Base
{
public:
	BaseB(){ std::cout << "BaseB Constructor\n"; }
	~BaseB(){ std::cout << "BaseB Destructor\n"; }
};


int main()
{
	BaseA* a = new BaseA();
	Base* base = dynamic_cast<Base*>(a);
	// 将会返回 nullptr
	BaseB* b = dynamic_cast<BaseB*>(base);
	// 将会输出
	if(!b)
		std::cout << "ERROR" << std::endl;
	
	
	return 0;
}
```

## reinterpret_cast

```cpp
int main()
{
	
	int x = 0x12345678;
	int* ptr = &x;
	// 指针 -> 整数（存储地址值）
	uintptr_t addr = reinterpret_cast<uintptr_t>(ptr);
	// 整数 -> 指针（危险，需确保地址有效）
	int* new_ptr = reinterpret_cast<int*>(addr);
	// 不相关类型的指针转换（语义上无意义，但语法允许）
	float* float_ptr = reinterpret_cast<float*>(ptr);
}
```
