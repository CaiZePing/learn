# 智能指针

## unique_ptr

```cpp
#include <iostream>
#include <string>
#include <memory>

class Entity
{
public:
	Entity()
	{
		std::cout << "Created Entity!" << std::endl;
	}
	~Entity()
	{
		std::cout << "Destroyed Entity!" << std::endl;
	}
	void Print(){}
};

int main()
{
	{
		// 出了这个作用域，将会自己 delete
		std::unique_ptr<Entity> entity = std::make_unique<Entity>();
		entity->Print();
	}
	return 0;
}
```

## share_ptr

```cpp
#include <iostream>
#include <string>
#include <memory>

class Entity
{
public:
	Entity()
	{
		std::cout << "Created Entity!" << std::endl;
	}
	~Entity()
	{
		std::cout << "Destroyed Entity!" << std::endl;
	}
	void Print(){}
};

int main()
{
	{
		std::share_ptr e;
		{
			// 知道这块内存没有引用，将会自动 delete
			std::share_ptr<Entity> entity = std::make_share<Entity>();
			e = entity;
		}
	}
	// 这边时，new 出来的内存会自动 delete
	return 0;
}
```

## weak_ptr

```cpp
#include <iostream>
#include <string>
#include <memory>

class Entity
{
public:
	Entity()
	{
		std::cout << "Created Entity!" << std::endl;
	}
	~Entity()
	{
		std::cout << "Destroyed Entity!" << std::endl;
	}
	void Print(){}
};

int main()
{
	{
		// weak 指针是弱指针，不会增加引用
		std::weak_ptr e;
		{
			// 知道这块内存没有引用，将会自动 delete
			std::share_ptr<Entity> entity = std::make_share<Entity>();
			e = entity;
		}
		// 这边时，new 出来的内存会自动 delete
		// 这时可以询问 e 的内存有没有效
	}
	
	return 0;
}
```
