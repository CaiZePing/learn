只是用来简化 `if` 语句的赋值操作，也可以优化一点点性能在声明变量时使用
三元运算符尽量不要嵌套使用

```cpp
#include <iostream>
#include <string>

static int s_Level = 1;
static int s_Speed = 2;

int main()
{
	if(s_Level > 5)
		s_Speed = 10;
	else
		s_Speed = 5;
	// 这种与上面的功能一摸一样	
	s_Speed = s_Level > 5 ? 10 : 5;
	
	std::string s_Temp = s_Level > 5 ? "Master" : "Beginner";
	// 下面这种的开销会比上面多一个默认构造
	std::string s_Temp2;
	if(s_Level > 5)
		s_Temp2 = "Master";
	else
		s_Temp2 = "Beginner";
	
	return 0;
}
```
