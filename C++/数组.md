数组实际上是一个指针

```cpp
#include <iostream>

int main()
{
	int example[5];
	int *ptr = example;
	// 将所有都赋值为 2
	for(int i = 0;i < 5;i++)
		example[i] = 2;

	*(ptr+2) = 5;

}
```

# vector 动态数组

## 基础用法

```cpp
#include <iostream>
#include <vector>

struct Vertex
{
	float x,y,z;
};

std::ostream& operator<<(std::ostream&stream,const Vertex&vertex)
{
	stream << vertex.x << ", " << vertex.y << ", " << vertex.z;
	return stream;
}
 
int main()
{
	std::vector<Vertex> vertices;
	// 在 main 中构造一个
	// 再复制过去
	vertices.push_back({1,2,3});
	vertices.push_back({3,4,5});	
	
	for(int i = 0;i < vertices.size();i++)
		std::cout << vertices[i] << std::endl;
	
	vertices.erase(vertices.begin()+1);
	
	for(Vertex& v : vertices)
		std::cout << v << std::endl;
	
	return 0;
}
```

## 简单优化

```cpp
#include <iostream>
#include <vector>

struct Vertex
{
	float x,y,z;
	Vertex(float x,float y,float z)
		: x(x),y(y),z(z){}
	Vertex(const Vertex&other)
		: x(other.x),y(other.y),z(other.z)
	{
		std::cout << "Copied!" << std::endl;
	}
};

std::ostream& operator<<(std::ostream&stream,const Vertex&vertex)
{
	stream << vertex.x << ", " << vertex.y << ", " << vertex.z;
	return stream;
}
 
int main()
{
	std::vector<Vertex> vertices;
	// 将申请8个存储 Vertex 的空间
	vertices.reserve(8);
	// 用参数去调用 Vertex 的构造函数
	// 直接在存储的地方构造
	vertices.emplace_back(1, 2, 3);
	vertices.emplace_back(1, 2, 3);
	vertices.emplace_back(1, 2, 3);
	vertices.emplace_back(1, 2, 3);
	
	return 0;
}
```

# array 静态数组

在大多数情况下与 `int arr[5]`  性能是一样的
但是因为 `array` 提供一些函数，所以使用起来要比原始的数组要方便

```cpp
#include <iostream>
#include <array>

int main()
{
	std::array<int,5> arr;
	arr[0] = 2;
	arr[4] = 3;
	// 根据不同的平台提供的哭
	// 有的库在调试情况下会报错（vs）
	// arr[5] = 4;
	std::cout << arr.Size() << std::endl;
}
```

# 多维数组

正常情况下不要使用堆分配多维数组
同样个数的情况下运行要比一维数组多很多时间

```cpp
#include <iostream>

int main()
{
	// 一维数组
	int *a1 = new int[5];
	// 二维数组
	int **a2 = new int*[5];
	for(int i = 0;i < 5;i++)
		a2[i] = new int[5];
	for(int x = 0;x < 5;x++)
		for(int y = 0;y < 5;y++)
			a2[x][y] = 2;
	// 三维数组
	int ***a3 = new int**[5];
	for(int i = 0;i < 5;i++)
	{
		a3[i] = new int*[5];
		for(int j = 0;j < 5;j++)
			a3[i][j] = new int[5];
	}
	for(int i = 0;i < 5;i++)
		for(int j = 0;j < 5;j++)
			for(int k = 0;k < 5;k++)
				a3[i][j][k] = 4;
	
	// 释放所有内存
	for(int i = 0;i < 5;i++)
	{
		for(int j = 0;j < 5;j++)
			delete[] a3[i][j];
		delete[] a3[i];
	}
	delete[] a3;
	
	for(int i = 0;i < 5;i++)
		delete[] a2[i];
	delete[] a2;
	
	delete[] array;
	
	return 0;
}
```
