# 函数使用

```cpp
#include <iostream>
#include <string>

template<typename T>
void Print(T value)
{
	std::cout << value << std::endl;
}

int main()
{
	Print(5);
	Print(4.3f);
	Print("Hello World");
	return 0;
}
```
