# tuple

```cpp
#include <iostream>
#include <tuple>

static std::tuple<std::string, std::string> func()
{
	std::string s1 = "Hello";
	std::string s2 = "World";
	
	return std::make_pair(s1, s2);
}

int main()
{
	auto s = func();
	
	std::cout << std::get<0>(s) << std::endl;
	std::cout << std::get<1>(s) << std::endl;
	
	return 0;
}
```

# pair

这个只能放两个变量

```cpp
#include <iostream>

static std::pair<std::string, std::string> func()
{
	std::string s1 = "Hello";
	std::string s2 = "World";
	
	return std::make_pair(s1, s2);
}

int main()
{
	auto s = func();
	
	std::cout << s.first << std::endl;
	std::cout << s.second << std::endl;
	
	return 0;
}
```

# 结构化绑定

C++ 17 及之后的特性，在使用时需要注意项目是不是 17 及更高

```cpp
#include <iostream>
#include <string>
#include <tuple>

std::tuple<std::string,int> CreatePerson()
{
	return { "Cai", 2 };
}

int main()
{
	auto[name,age] = CreatePerson();
	std::cout << name << ", " << age << std::endl;
	return 0;
}
```
